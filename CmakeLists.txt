cmake_minimum_required(VERSION 3.1...3.25)

project(
	Opengl
	VERSION 0.1.0
	LANGUAGES CXX C
)
find_package(glad CONFIG REQUIRED)
find_package(glfw3 CONFIG REQUIRED)
find_package(imgui CONFIG REQUIRED)
find_package(glm CONFIG REQUIRED)

add_executable(App  "source/main.cpp"  "source/main.h" "source/shader.h" 
"source/shader.cpp" "source/object.cpp" "source/object.h" "source/VertexBuffer.h" 
"source/VertexBuffer.cpp" "source/IndexBuffer.cpp" "source/IndexBuffer.h")

target_link_libraries(App PRIVATE glad::glad imgui::imgui glfw glm::glm)

target_compile_definitions(App PRIVATE
		RESOURCE_DIR="${CMAKE_CURRENT_SOURCE_DIR}/resources"
	)
