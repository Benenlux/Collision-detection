cmake_minimum_required(VERSION 3.1...3.25)

project(
	Opengl
	VERSION 0.1.0
	LANGUAGES CXX C
)
set(CMAKE_CXX_STANDARD 17)
find_package(glad CONFIG REQUIRED)
find_package(glfw3 CONFIG REQUIRED)
find_package(imgui CONFIG REQUIRED)
find_package(glm CONFIG REQUIRED)

set(graphics "${CMAKE_CURRENT_SOURCE_DIR}/source/graphics")
set(physics "${CMAKE_CURRENT_SOURCE_DIR}/source/physics")
set(SOURCES source/main.cpp ${graphics}/IndexBuffer.cpp ${graphics}/shader.cpp ${graphics}/VertexArray.cpp ${graphics}/VertexBuffer.cpp "source/graphics/Object.cpp" "source/physics/RigidBody.cpp" "source/Scene.cpp")
set(HEADERS source/main.h ${graphics}/IndexBuffer.h ${graphics}/shader.h ${graphics}/VertexArray.h ${graphics}/VertexBuffer.h ${graphics}/VertexBufferLayout.h "source/graphics/Object.h" ${physics}/RigidBody.h "source/Scene.h" "source/physics/Colliders.h")

add_executable(App  ${SOURCES} ${HEADERS} "source/physics/RigidBody.cpp")

target_link_libraries(App PRIVATE glad::glad imgui::imgui glfw glm::glm)

target_compile_definitions(App PRIVATE
		RESOURCE_DIR="${CMAKE_CURRENT_SOURCE_DIR}/resources"
	)
